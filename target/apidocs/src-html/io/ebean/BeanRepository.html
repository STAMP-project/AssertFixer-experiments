001package io.ebean;
002
003import io.ebean.bean.EntityBean;
004
005/**
006 * Provides finder functionality for use with "Dependency Injection style" use of Ebean.
007 * <p>
008 * <pre>{@code
009 *
010 * @Repository
011 * public class CustomerRepository extends BeanRepository<Long,Customer> {
012 *
013 *   @Inject
014 *   public CustomerRepository(EbeanServer server) {
015 *     super(Customer.class, server);
016 *   }
017 *
018 *   // ... add customer specific finders and persist logic
019 *
020 *   public List<Customer> findByName(String nameStart) {
021 *     return query().where()
022 *             .istartsWith("name", nameStart)
023 *             .findList();
024 *   }
025 *
026 * }
027 * }</pre>
028 *
029 * @param <I> The ID type
030 * @param <T> The Bean type
031 */
032public abstract class BeanRepository<I, T> extends BeanFinder<I, T> {
033
034  /**
035   * Create with the given bean type and EbeanServer instance.
036   * <p>
037   * Typically users would extend BeanRepository rather than BeanFinder.
038   * </p>
039   * <pre>{@code
040   *
041   *   @Inject
042   *   public CustomerRepository(EbeanServer server) {
043   *     super(Customer.class, server);
044   *   }
045   *
046   * }</pre>
047   *
048   * @param type   The bean type
049   * @param server The EbeanServer instance typically created via Spring factory or equivalent
050   */
051  protected BeanRepository(Class<T> type, EbeanServer server) {
052    super(type, server);
053  }
054
055  /**
056   * Marks the entity bean as dirty.
057   * <p>
058   * This is used so that when a bean that is otherwise unmodified is updated the version
059   * property is updated.
060   * <p>
061   * An unmodified bean that is saved or updated is normally skipped and this marks the bean as
062   * dirty so that it is not skipped.
063   * <p>
064   * <pre>{@code
065   *
066   * Customer customer = customerRepository.byId(id);
067   *
068   * // mark the bean as dirty so that a save() or update() will
069   * // increment the version property
070   *
071   * customerRepository.markAsDirty(customer);
072   * customerRepository.save(customer);
073   *
074   * }</pre>
075   *
076   * @see EbeanServer#markAsDirty(Object)
077   */
078  public void markAsDirty(T bean) {
079    db().markAsDirty(bean);
080  }
081
082  /**
083   * Mark the property as unset or 'not loaded'.
084   * <p>
085   * This would be used to specify a property that we did not wish to include in a stateless update.
086   * </p>
087   * <pre>{@code
088   *
089   *   // populate an entity bean from JSON or whatever
090   *   Customer customer = ...;
091   *
092   *   // mark the email property as 'unset' so that it is not
093   *   // included in a 'stateless update'
094   *   customerRepository.markPropertyUnset(customer, "email");
095   *
096   *   customerRepository.update(customer);
097   *
098   * }</pre>
099   *
100   * @param propertyName the name of the property on the bean to be marked as 'unset'
101   */
102  public void markPropertyUnset(T bean, String propertyName) {
103    ((EntityBean) bean)._ebean_getIntercept().setPropertyLoaded(propertyName, false);
104  }
105
106  /**
107   * Insert or update this entity depending on its state.
108   * <p>
109   * Ebean will detect if this is a new bean or a previously fetched bean and perform either an
110   * insert or an update based on that.
111   *
112   * @see EbeanServer#save(Object)
113   */
114  public void save(T bean) {
115    db().save(bean);
116  }
117
118  /**
119   * Update this entity.
120   *
121   * @see EbeanServer#update(Object)
122   */
123  public void update(T bean) {
124    db().update(bean);
125  }
126
127  /**
128   * Insert this entity.
129   *
130   * @see EbeanServer#insert(Object)
131   */
132  public void insert(T bean) {
133    db().insert(bean);
134  }
135
136  /**
137   * Delete this bean.
138   * <p>
139   * This will return true if the bean was deleted successfully or JDBC batch is being used.
140   * </p>
141   * <p>
142   * If there is no current transaction one will be created and committed for
143   * you automatically.
144   * </p>
145   * <p>
146   * If the Bean does not have a version property (or loaded version property) and
147   * the bean does not exist then this returns false indicating that nothing was
148   * deleted. Note that, if JDBC batch mode is used then this always returns true.
149   * </p>
150   *
151   * @see EbeanServer#delete(Object)
152   */
153  public boolean delete(T bean) {
154    return db().delete(bean);
155  }
156
157  /**
158   * Delete a bean permanently without soft delete.
159   * <p>
160   * This is used when the bean contains a <code>@SoftDelete</code> property and we
161   * want to perform a hard/permanent delete.
162   * </p>
163   *
164   * @see EbeanServer#deletePermanent(Object)
165   */
166  public boolean deletePermanent(T bean) {
167    return db().deletePermanent(bean);
168  }
169
170  /**
171   * Merge this entity using the default merge options.
172   * <p>
173   * Ebean will detect if this is a new bean or a previously fetched bean and perform either an
174   * insert or an update based on that.
175   *
176   * @see EbeanServer#merge(Object)
177   */
178  public void merge(T bean) {
179    db().merge(bean);
180  }
181
182  /**
183   * Merge this entity using the specified merge options.
184   * <p>
185   * Ebean will detect if this is a new bean or a previously fetched bean and perform either an
186   * insert or an update based on that.
187   *
188   * @see EbeanServer#merge(Object, MergeOptions)
189   */
190  public void merge(T bean, MergeOptions options) {
191    db().merge(bean, options);
192  }
193
194  /**
195   * Refreshes this entity from the database.
196   *
197   * @see EbeanServer#refresh(Object)
198   */
199  public void refresh(T bean) {
200    db().refresh(bean);
201  }
202}
